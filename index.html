Example Javascript 👩‍💻

async function increment(token, username = "Unnamed Player", number = 1) { let r = await fetch('https://gsapi.ndev.tk/api', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ token: token, username: username, increment: number }) }); return r.json(); }
